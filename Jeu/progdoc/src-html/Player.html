001    import java.util.Stack;
002    
003    public class Player 
004     {
005              private String playerName;
006              private int playerCurrentWeight,playerMaxWeight,playerWeight;
007              private Room currentRoom ;
008              private Stack<Room> walkthroughRoom ;
009              private ItemList inventory ;
010              
011    
012    
013           public Player(String pPlayerName,int pPlayerMaxWeight, Room pCurrentRoom)
014          {   this.playerWeight=20;
015              this.playerName= pPlayerName;
016              this.playerMaxWeight= pPlayerMaxWeight;
017              this.playerCurrentWeight= 20;
018              this.currentRoom= pCurrentRoom;
019              this.walkthroughRoom = new Stack<Room>();
020              this.inventory = new ItemList();
021         }
022         
023          public int getPlayerWeight()
024          { return this.playerWeight; }
025       
026           public String getPlayerName()
027          {   return this.playerName; }
028          
029          public int getPlayerCurrentWeight()
030          {    return this.playerCurrentWeight;}
031          
032          public Room getCurrentRoom()
033          {   return this.currentRoom; }
034          
035          public Room setCurrentRoom(Room pCurrentRoom)
036       
037          {    currentRoom = pCurrentRoom;
038               return this.currentRoom;
039            }
040          
041          public int getPlayerMaxWeight()
042          {    return this.playerMaxWeight;}
043          
044          public void setPlayerMaxWeight(int pAddMaxWeight)
045          {   playerMaxWeight += pAddMaxWeight; }
046    
047          public Stack<Room> getWalkthroughRoom()
048          {    return this.walkthroughRoom;}
049          
050          /**
051             */
052          public boolean lookWalkthroughRoom()
053          {    return this.walkthroughRoom.empty(); }
054    
055          /**
056           */
057            public void actualiseWalkthroughRoom()
058           {   walkthroughRoom.push(currentRoom); }
059    
060           public Room getPreviousRoom()
061           {   return this.walkthroughRoom.pop();   }
062           
063           public void removeWalkthroughRoom()
064           { while (this.walkthroughRoom.empty() == false )
065               this.walkthroughRoom.pop();
066            
067            
068            }
069    
070           public boolean canBePickedUp(String pItemName)
071            {
072            if ( this.playerCurrentWeight + currentRoom.getItem(pItemName).getWeight()> playerMaxWeight) 
073                 return false;
074            else
075                 return true;
076                                 }
077    
078           public boolean canBeDroped(String pItemName)
079            {
080            if (inventory.getItem(pItemName) ==null)
081                    return false;
082             else  
083                     return true;
084                                  }
085             
086             
087             public void addItemInventory(String pItemName)
088             {
089                 playerCurrentWeight=playerCurrentWeight+currentRoom.getItem(pItemName).getWeight();
090                 inventory.addItem(currentRoom.getItem(pItemName));
091                                   }
092        
093                public void removeItem(String pItemName)
094            {
095                playerCurrentWeight = playerCurrentWeight - inventory.getItem(pItemName).getWeight();
096                inventory.removeItem(currentRoom.getItem(pItemName).getNameItem());
097                                            }
098                                            
099           public ItemList getInventory()
100           {         return inventory;     }
101           
102           
103           
104    }
