021

026               
027            }
028            
029         String takeSecondWord = getSecondWord();
030         
031             if(aGameModel.getCurrentRoom().getItem(takeSecondWord) == null)

033                 aUserInterface.println("LSD is bad for health.");
034               
035             }
036
