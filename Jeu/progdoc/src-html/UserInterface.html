075        /**
076         * Show an image file in the interface.
077         */
078        public void showImage(String imageName)
079        {
080            URL imageURL = this.getClass().getClassLoader().getResource("Pictures/"+imageName);
081            if(imageURL == null)
082                System.out.println( "image not found");
083            else {
084                ImageIcon icon = new ImageIcon(imageURL);
085                image.setIcon(icon);
086                myFrame.pack();
087            }
088        }

234        /**
235         * Actionlistener interface for entry textfield.
236         */
237        public void actionPerformed(ActionEvent e) 
238        {
239           
240            
241            Object objet = e.getSource();
242            if (objet == entryField )
243               processCommand();
244            else if(objet == buttonRagequit )
245            processCommand( "quit");
246            else if(objet == buttonLook)
247            processCommand( "look") ;
248            else if(objet == buttonPrevious)
249            processCommand("back");
250            else if(objet == buttonInventory)
251            processCommand("items");
252            else if(objet == buttonHelp)
253            processCommand("help");
254            
255            else if(objet == buttonRight)
256            processCommand("go right");
257            else if(objet == buttonStraight)
258            processCommand("go straight");
259            else if(objet == buttonLeft)
260            processCommand("go left");
261            else if(objet == buttonBack)
262            processCommand("go back");
263            
264            else if(objet == buttonUp)
265            processCommand("go up");
266            else if(objet == buttonRun)
267            processCommand("go run");
268            else if(objet == buttonDown)
269            processCommand("go down");
270            
271            else if (objet==realTime)
272              aGameEngine.getTime().decrement();
273             if ( aGameEngine.getCountDownState() ==false )
274             { realTime.stop();
275               countdownLabel.setText(aGameEngine.getTime().counterString());
276               aGameEngine.endGame() ;}
277               else
278               countdownLabel.setText(aGameEngine.getTime().counterString());
279            
280        }
281    
282       private void processCommand()
283       {
284          String input = entryField.getText();
285          entryField.setText( "");
286          aGameEngine.interpretCommand(input);
287        }
288        
289       private void processCommand(String bouton)
290       { 
291          aGameEngine.interpretCommand(bouton);
292         }
