001    import java.util.Iterator;
002    import java.util.HashMap;
003    import java.util.Set;
004    /**
005     * This class is part of the "World of Zuul" application. 
006     * "World of Zuul" is a very simple, text based adventure game.  
007     * 
008     * This class holds an enumeration of all command words known to the game.
009     * It is used to recognise commands as they are typed in.
010     *
011     * @author  Michael Kolling and David J. Barnes
012     * @version 2006.03.30
013     */
014    
015    public class CommandWords
016    {
017          
018        private HashMap<String, Command> validCommands;
019    
020        /**
021         * Constructor - initialise the command words.
022         */
023        public CommandWords()
024        {
025           validCommands = new HashMap<String, Command>();
026            validCommands.put("help", new HelpCommand());
027            validCommands.put("go", new GoCommand());
028            validCommands.put("look", new LookCommand());
029            validCommands.put("eat", new EatCommand());
030            validCommands.put("back", new BackCommand());
031            validCommands.put("take", new TakeCommand());
032            validCommands.put("drop", new DropCommand());
033            validCommands.put("quit", new QuitCommand());
034            validCommands.put("test", new TestCommand());
035            validCommands.put("items", new ItemsCommand());
036            validCommands.put("use", new UseCommand());
037            validCommands.put("examine", new ExamineCommand());
038            validCommands.put("alea",new AleaCommand()) ;
039            
040            
041        }
042        
043        
044        public Command getCommandWord(String commandWord)
045        {
046            return (Command)validCommands.get(commandWord);
047         
048        }
049        
050        /**
051         * Check whether a given String is a valid command word. 
052         * @return true if a given string is a valid command,
053         * false if it isn't.
054         */
055        public boolean isCommand(String aString)
056        {
057            return validCommands.containsKey(aString);
058        }
